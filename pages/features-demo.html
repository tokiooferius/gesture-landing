🔧 Integration Example
Using Training Mode:
import TrainingMode from '../src/services/trainingMode.js';

// Initialize
const training = new TrainingMode();

// Start lesson
training.startLesson(1);

// Record correct detection
const result = training.recordCorrectDetection('ok');
// → { streak: 1, score: 10, totalScore: 10, bonus: '' }

// Get stats
const stats = training.getStats();
// → { totalScore, currentStreak, nextMilestone, ... }

// Get achievements
const badges = training.getAchievements();
// → [{ id, name, icon }, ...]
